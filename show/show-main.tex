\section{研究背景与问题动机}

\begin{frame}{在线图推理的效率瓶颈}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\highlight{问题一：推理复杂度爆炸}}
      \begin{itemize}
        \item $L$ 层 GCN 单次前向主项复杂度：$O(L\cdot m\cdot d)$，$m=\mathrm{nnz}(A)$
        \item 有效感受野随层数指数膨胀：\highlight{$O(\bar{d}^L)$ 个节点}
        \item 3层、$\bar{d}=10$：需聚合 $\approx1000$ 个邻居
        \item 在线预测时延可超过 \highlight{100\,ms}（实时系统不可接受）
      \end{itemize}

      \vspace{0.25cm}
      \textbf{\highlight{问题二：加深网络的两大病态}}
      \begin{itemize}
        \item \blue{过平滑（Over-smoothing）}：节点表示趋同，信息区分度丧失
        \item \blue{过压缩（Over-squashing）}：远端信息丢失，多跳瓶颈效应
        \item 简单加深层数无法获得更好的感受野-效率平衡
      \end{itemize}

      \vspace{0.25cm}
      \textbf{已有代表性方向及不足}
      \begin{itemize}
        \item 邻居采样（GraphSAGE、FastGCN）：推理仍需访图
        \item 预计算线性化（SGC、APPNP）：缺节点级自适应
        \item 稀疏分解（\highlight{SDGNN}）：本文直接扩展基线
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      \begin{alertblock}{在线服务场景的刚性约束}
        \begin{itemize}
          \item 推荐系统：\highlight{QPS $>$ 1000}，毫秒级响应
          \item 实时风控：延迟 \highlight{$<$ 10\,ms}
          \item 大规模图（$n>10^6$）：内存访问成为主瓶颈
        \end{itemize}
      \end{alertblock}

      \includegraphics[width=\linewidth]{figures/3-2.pdf}
      \vspace{0.1cm}
      \centering\footnotesize 图：SDGNN全局统一稀疏预算失效示意
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{SDGNN 的突破与局限——本文的问题来源}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{\green{SDGNN 核心思想}（Hu et al., 2024）}
        \small
        以单次稀疏分解替代多层消息传递，推理阶段仅执行：
        \begin{equation*}
          \hat{H}=\Theta^{\top}\tilde{\Phi}
        \end{equation*}
        $\tilde{\Phi}\in\mathbb{R}^{n\times d}$：预缓存特征；$\Theta\in\mathbb{R}^{n\times n}$：离线固化稀疏权重\\[3pt]
        主项复杂度：$O(L\cdot m\cdot d)\ \longrightarrow$ \highlight{$O(n\kbar d)$}\\[2pt]
        ogbn-arxiv 批次前向：\blue{9.2\,ms}，相对 GCN 加速 $\mathbf{13.6\times}$
      \end{block}

      \vspace{0.2cm}
      \textbf{\highlight{核心局限：全局统一稀疏预算}}
      \begin{itemize}
        \item 以同一 $\lambda_{\mathrm{reg}}$ 约束\textbf{所有节点}
        \item \textbf{枢纽节点}（高度数）：稀疏化过强 $\to$ 丢失关键邻居
        \item \textbf{外围节点}（低度数）：稀疏化不足 $\to$ 保留冗余连接
        \item 复杂图（异配、重尾度）：失配更明显
      \end{itemize}
      \vspace{0.15cm}
      \textbf{本文问题}：能否在保持 $O(n\kbar d)$ 接口不变的前提下，\\
      引入\textbf{节点级差异化稀疏预算}？
    \end{column}

\section{问题形式化与图复杂度指标}

\begin{frame}{第3章：问题形式化与优化目标}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{\blue{在 SDGNN 接口内引入节点级正则}}

      \textbf{训练阶段}（对每个节点 $i$ 求解带节点级正则的加权 Lasso）：
      \begin{equation*}
        \min_{\theta_i}\;
        \bigl\|h_i^* - \tilde{\Phi}^{\top}\theta_i\bigr\|_2^2
        +\tau(i)\|\theta_i\|_1
        \;\text{s.t.}\;
        \|\theta_i\|_0 \leq \textstyle\sum_{l=1}^{L}k_i^{(l)}
      \end{equation*}

      \textbf{推理阶段}（$\Theta^{\mathrm{fixed}}$ 离线固化后）：
      \begin{equation*}
        \hat{H}=\bigl(\Theta^{\mathrm{fixed}}\bigr)^{\top}\tilde{\Phi}_{\mathrm{cache}}
        \quad\Rightarrow\quad \highlight{O(n\kbar d)}
      \end{equation*}

      \vspace{0.2cm}
      \textbf{\blue{三项设计准则}}
      \begin{itemize}
        \item \textbf{可解释性}：$\tau(i)$ 与 $k_i^{(l)}$ 由可计算统计量确定，参数意义明确
        \item \textbf{复杂度不变}：在线推理严格保持 $O(n\kbar d)$
        \item \textbf{任务无关性}：不依赖标签统计量，等权实例化
      \end{itemize}

      \vspace{0.2cm}
      \textbf{表示近似误差}：
      $\;\mathcal{E}_{\mathrm{approx}} = \frac{1}{\sqrt{n}}\|H^*-\hat{H}\|_F$
    \end{column}

    \begin{column}{0.46\textwidth}
      \includegraphics[width=\linewidth]{figures/3-1.pdf}
      \vspace{0.05cm}
      \centering\footnotesize 图3.1：第3章整体流程图——从图输入到四类复杂度指标输出
    \end{column}
  \end{columns}
\end{frame}

% ==============================================================
\begin{frame}{第3章：四类图复杂度指标——自适应参数的统一输入接口}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\blue{1 \; 谱间隙 $\lambda_{\mathrm{gap}}$}}（全图级·频率维）
      \begin{itemize}
        \item 归一化拉普拉斯第二小特征值，刻画代数连通度
        \item Lanczos 迭代谱估计，$O(m\cdot t)$
        \item $\lambda_{\mathrm{gap}}$ 大 $\Rightarrow$ 连通性强 $\Rightarrow$ 近跳信息充足
        \item 驱动分层跳距保留率 $p_l$ 的整体设计
      \end{itemize}

      \vspace{0.2cm}
      \textbf{\blue{2 \; 节点谱能量 $E_{\mathrm{spectral}}(i)$}}（节点级·频率维）
      \begin{equation*}
        E_{\mathrm{spectral}}(i) = \sum_{k=1}^{K}w_k\,|U_{ik}|^2,\quad
        w_k=\tfrac{1}{K}\ (\text{等权})
      \end{equation*}
      \begin{itemize}
        \item 节点 $i$ 在各谱分量上的参与强度
        \item Lanczos 近似特征向量投影，$O(K\cdot m)$
        \item 直接进入 $\tau(i)$ 多因子融合
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{\blue{3 \; 归一化局部图熵 $H_{\mathrm{norm}}(i)$}}（节点维）
      \begin{equation*}
        H_{\mathrm{norm}}(i)
        =-\frac{1}{\log d_i}\sum_{j\in\mathcal{N}(i)}\tfrac{1}{d_i}\log\tfrac{1}{d_i}
      \end{equation*}
      \begin{itemize}
        \item 节点局部邻域的信息丰富程度
        \item $H_{\mathrm{norm}}\uparrow\ \Rightarrow$ 邻域多样 $\Rightarrow$ $\tau(i)$ 偏小
      \end{itemize}

      \vspace{0.2cm}
      \textbf{\blue{4 \; 度中心性 $C_{\mathrm{deg}}(i)$ 与 $k$-core 指数}}（节点维）
      \begin{equation*}
        C_{\mathrm{deg}}(i)=\frac{d_i}{n-1},\qquad
        \mathrm{core}(i)\in\mathbb{Z}^+
      \end{equation*}
      \begin{itemize}
        \item 节点在全局连接中的枢纽地位
        \item 枢纽节点（$C_{\mathrm{deg}}\uparrow$、$\mathrm{core}\uparrow$）：$\tau(i)$ 应\textbf{偏小}
        \item 外围节点：局部冗余多，$\tau(i)$ 应\textbf{偏大}
      \end{itemize}

      \vspace{0.1cm}
      \begin{alertblock}{四类指标的接口关系}
        \small 全部四类统计量在第4章统一进入自适应参数设计；在第5章通过分层LARS进入稀疏权重学习流程
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{三维自适应稀疏机制}

\begin{frame}{第4章：三维自适应稀疏机制整体框架}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \includegraphics[width=\linewidth]{figures/4-1.pdf}
      \centering\footnotesize 图4.1：三维自适应稀疏机制整体框架图
    \end{column}

    \begin{column}{0.50\textwidth}
      \textbf{\blue{频率维（§\,4.1）}}：放松"统一低通传播"假设
      \begin{itemize}
        \item 节点谱能量 $E_{\mathrm{spectral}}(i)$ 作为\textbf{等权统一谱输入} $w_k(i)$
        \item 任务无关，不引入标签统计量或同配性假设
      \end{itemize}

      \vspace{0.2cm}
      \textbf{\highlight{节点维（§\,4.2）}}：放松"统一稀疏压力"假设
      \begin{itemize}
        \item 多因子融合构造节点级正则系数 $\tau(i)$
        \item 输入：$E_{\mathrm{spectral}}(i)$、$H_{\mathrm{norm}}(i)$、$C_{\mathrm{deg}}(i)$、$\mathrm{core}(i)$
        \item 结构信息贡献高的节点 $\Rightarrow$ $\tau(i)$ 较小 $\Rightarrow$ 保留更多邻居
      \end{itemize}

      \vspace{0.2cm}
      \textbf{\green{跳距维（§\,4.3）}}：放松"各跳贡献均等"假设
      \begin{itemize}
        \item 近邻优先分层预算 $k_i^{(l)}$，保留率随跳距递减
        \item $p_l=p_{\mathrm{base}}\cdot\varrho^{l-1}$（$\varrho\in(0,1)$），$k_i^{(l)}=\lfloor p_l\cdot|\mathcal{R}_l(i)|\rfloor$
        \item §\,4.3.1 多跳传播信息衰减界分析提供理论依据
      \end{itemize}

      \vspace{0.15cm}
      \begin{alertblock}{关键约束}
        \small 三个维度均不改变推理接口：
        $\hat{H}=(\Theta^{\mathrm{fixed}})^{\top}\tilde{\Phi}_{\mathrm{cache}}$，
        主项复杂度 \highlight{$O(n\kbar d)$} 严格保持不变
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

% S09  第4章：τ(i) 具体构造

\begin{frame}{节点级正则系数 $\tau(i)$ 的多因子融合构造（§\,4.2）}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{\blue{多因子融合公式}}
      \begin{equation*}
        \tau(i)=\tau_{\mathrm{base}}\cdot
        \exp\!\bigl(-\beta_\tau\,E_{\mathrm{spectral}}(i)\bigr)
        \cdot
        f\!\bigl(H_{\mathrm{norm}}(i),\;C_{\mathrm{deg}}(i),\;\mathrm{core}(i)\bigr)
      \end{equation*}
      其中 $f(\cdot)$ 为单调递减映射。

      \vspace{0.2cm}
      \textbf{各因子贡献方向（均为单调递减关系）}
      \begin{itemize}
        \item $E_{\mathrm{spectral}}(i)\uparrow$ $\Rightarrow$ 高频参与强 $\Rightarrow$ \highlight{$\tau(i)\downarrow$}（保留更多）
        \item $H_{\mathrm{norm}}(i)\uparrow$ $\Rightarrow$ 邻域多样 $\Rightarrow$ \highlight{$\tau(i)\downarrow$}
        \item $C_{\mathrm{deg}}(i)\uparrow$ $\Rightarrow$ 枢纽节点 $\Rightarrow$ \highlight{$\tau(i)\downarrow$}
        \item $\mathrm{core}(i)\uparrow$ $\Rightarrow$ 核心层级高 $\Rightarrow$ \highlight{$\tau(i)\downarrow$}
      \end{itemize}

      \vspace{0.2cm}
      \textbf{超参数经验稳健区间}（Cora 验证集，10次随机种子）
      \begin{itemize}
        \item $\beta_\tau\in[0.8,1.2]$：准确率波动 $\leq 0.3\%$
        \item $\tau_{\mathrm{base}}\in[5\times10^{-4},5\times10^{-3}]$：87.7\%--88.0\%
        \item B2-RND（随机化 $\tau$ 对照）未带来稳定收益，证明结构驱动有效
      \end{itemize}
    \end{column}

\section{算法整合与理论分析}

\begin{frame}{分层跳距预算与 LARS 算法整合（§\,4.3 + §\,5.1）}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\green{分层跳距预算 $k_i^{(l)}$}}（§\,4.3）

      \textbf{理论依据}（§\,4.3.1 信息衰减界分析）：多跳传播中近跳贡献更大，信息边际贡献随 $l$ 单调递减

      近邻优先保留率（参考分配策略）：
      \begin{equation*}
        p_l=p_{\mathrm{base}}\cdot\varrho^{l-1},\quad\varrho\in(0,1)
      \end{equation*}
      \begin{equation*}
        k_i^{(l)}=\lfloor p_l\cdot|\mathcal{R}_l(i)|\rfloor\leq|\mathcal{R}_l(i)|
      \end{equation*}

      \vspace{0.15cm}
      \textbf{跳距策略消融验证}（Cora，表\,6.4）：
      \begin{center}\small
      \begin{tabular}{lcc}
        \toprule
        策略 & 准确率 & $\mathcal{E}_{\mathrm{approx}}$\\
        \midrule
        近邻优先（参考分配） & \green{\textbf{88.0\%}} & \green{\textbf{0.151}}\\
        均匀分配 & 87.2\% & 0.182\\
        反向分配（深层更多） & \highlight{86.5\%} & \highlight{0.218}\\
        \bottomrule
      \end{tabular}
      \end{center}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{\blue{LARS 分层稀疏求解与交替优化}}（算法\,5.1/5.2）

      \vspace{0.1cm}
      \textbf{训练阶段（算法\,5.1）}
      \begin{itemize}
        \item \textbf{输入}：图 $G$，教师表示 $H^*$，$\tau(i)$，$k_i^{(l)}$
        \item \textbf{Phase $W$}：固定 $\Theta$，更新特征变换参数 $W$
        \item \textbf{Phase $\Theta$}：固定 $W$，对每节点 $i$ 执行分层 LARS，以 $\tau(i)$ 为正则，$k_i^{(l)}$ 为截断上界
        \item \textbf{输出}：固化 $\Theta^{\mathrm{fixed}}$，缓存 $\tilde{\Phi}_{\mathrm{cache}}$
      \end{itemize}

      \vspace{0.15cm}
      \textbf{推理阶段（算法\,5.2）}
      \begin{itemize}
        \item 输入 $\Theta^{\mathrm{fixed}}$，$\tilde{\Phi}_{\mathrm{cache}}$
        \item 执行：$\hat{H}=(\Theta^{\mathrm{fixed}})^{\top}\tilde{\Phi}_{\mathrm{cache}}$
        \item 主项复杂度：\highlight{$O(n\kbar d)$}（单次稀疏矩阵乘）
        \item 无需访图，无需多跳邻域展开
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{第5章：条件化误差分析与复杂度全景（§\,5.2--5.3）}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{\blue{条件化误差分析}}（§\,5.2）

      \textbf{前提条件（$\sigma$-谱相似性）}：稀疏图 $\tilde{G}$ 满足对原图 $G$ 的 $\sigma$-近似，即对任意向量 $x$：
      \begin{equation*}
        (1-\sigma)x^{\top}Lx\;\leq\;x^{\top}\tilde{L}x\;\leq\;(1+\sigma)x^{\top}Lx
      \end{equation*}

      \textbf{结论}：在此条件下，$\mathcal{E}_{\mathrm{approx}}$ 可由稀疏支撑质量与 $\lambda_{\mathrm{gap}}$ 控制，为第4章自适应参数设计提供\textbf{误差方向解释}

      \vspace{0.1cm}
      \begin{alertblock}{诚实说明（§\,7.3）}
        \small 该分析为条件化解释，非对所有图结构最优稀疏性的严格保证
      \end{alertblock}

\section{总结与展望}

\begin{frame}{主要贡献与创新点}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\blue{贡献一}}：在 SDGNN 接口内引入节点级自适应稀疏分配
      \begin{itemize}
        \item 以可计算图结构统计量（谱间隙、谱能量、局部图熵、中心性）驱动"哪个节点保留多少邻居"的参数化设计
        \item 将统一稀疏控制问题转化为具体的节点级参数设计问题
        \item 保留单次稀疏矩阵乘推理形式与 $O(n\kbar d)$ 复杂度保证
      \end{itemize}

      \vspace{0.2cm}
      \textbf{\blue{贡献二}}：构建三维自适应参数体系
      \begin{itemize}
        \item \textbf{频率维}：等权节点谱能量（任务无关）
        \item \textbf{节点维}：多因子融合 $\tau(i)$（谱能量+图熵+度中心性+$k$-core）
        \item \textbf{跳距维}：近邻优先分层预算 $k_i^{(l)}$（随跳距递减）
        \item 三个维度在第5章共同进入分层 LARS 求解流程，图结构统计量从解释变量转化为稀疏权重学习的实际调节因素
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{\blue{贡献三}}：在统一实验口径下验证精度与效率

      \textbf{精度}（vs.\ SDGNN，Wilcoxon $p<0.05$，全部6个数据集）：
      \begin{itemize}
        \item 引文网络均值 \green{+1.3\%}，结果方差同步下降
        \item 网页图均值 \green{+1.9\%}，复杂拓扑下更优
        \item ogbn-arxiv \green{+0.86\%}，大规模图上有效
      \end{itemize}

      \vspace{0.15cm}
      \textbf{效率}（推理阶段，相对 GCN）：
      \begin{itemize}
        \item 四数据集推理加速 \blue{$8.0\times$}（均值）
        \item ogbn-arxiv 加速 \blue{$14.7\times$}
        \item 峰值显存节省 \blue{74.8\%}
        \item 相对 SDGNN 额外显存增量仅 +41\,MB
      \end{itemize}

      \vspace{0.15cm}
      \begin{alertblock}{方法适用范围}
        \small 教师表示已给定、稀疏权重可离线固化、在线推理需低时延的 SDGNN 式稀疏分解场景
      \end{alertblock}
    \end{column}
  \end{columns}

\begin{frame}{研究局限与未来展望}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{\highlight{研究局限性}}（§\,7.3）
      \begin{enumerate}\setlength{\itemsep}{4pt}
        \item \textbf{启发式近似性质}\\
          $\tau(i)$ 基于代理变量构造，尚未严格刻画与最优稀疏支撑的充要关系；重尾度分布、强异配图上的适用边界待进一步分析
        \item \textbf{静态图假设}\\
          方法适用 transductive 设定，$\Theta^{\mathrm{fixed}}$ 在新节点或边结构变化时未必继续适用；动态场景的稀疏权重维护未系统讨论
        \item \textbf{误差界的条件化性质}\\
          谱相似性分析提供误差\textbf{方向解释}，非对所有图结构最优稀疏性的严格保证
      \end{enumerate}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{\blue{未来研究方向}}（§\,7.4）
      \begin{enumerate}\setlength{\itemsep}{4pt}
        \item \textbf{理论深化}\\
          引入局部谱扰动分析，建立 $\tau(i)$、$k_i^{(l)}$ 与 $\sigma$-谱相似性的可检验关系，将条件化误差解释推进为直接参数设计指导
        \item \textbf{结构可靠性增强}\\
          引入边级可靠性评分（结构相似性+对比学习表征），通过软门控调节 LARS 候选集，联合优化图结构质量与预算分配
        \item \textbf{动态图适配}\\
          设计增量维护机制，局部变化时对受影响节点进行 warm-start 重估；实现分层更新策略（局部微调/部分重训练/全局热启动）
      \end{enumerate}
